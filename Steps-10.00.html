Add Calendar

  1. In Power BI under Modeling Tab click on New Table 
  2. Paste the following DAX code in the formula bar 
  3. Press Enter and Save the file

Date =

    VAR MinDate = YEAR ( MIN ( Shooting[date] ) )

    VAR MaxDate = YEAR ( MAX ( Shooting[date] ) )

 

    RETURN

    ADDCOLUMNS(

        FILTER(

            CALENDARAUTO(),

            YEAR ( [Date] ) >= MinDate &&

            YEAR ( [Date] ) <= MaxDate

        ),

    "Year" , YEAR( [Date] ),

    "Month" , FORMAT( [Date], "mmm" ) ,

    "MonthNum" , MONTH( [Date] ),

    "Quarter" , FORMAT( [Date], "\QQ" ),

    "DayOfMonth" , FORMAT ( [Date] , "dd" )    )
